def process_telemetry_data():
    # Extract from On-Prem Source
    raw_df = spark.read.format("jdbc") \
        .option("url", jdbc_url) \
        .load()
    
    # Apply Transformations & Cleansing
    clean_df = raw_df.filter(col("status") == "Active") \
        .withColumn("processed_at", current_timestamp())
    
    # Load to ADLS (Delta format)
    clean_df.write.format("delta") \
        .mode("append") \
        .saveAsTable("trusted_zone.telemetry")
        
    return "ETL Job Completed"
